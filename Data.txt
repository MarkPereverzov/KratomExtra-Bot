CREATE TABLE User (ID INT PRIMARY KEY, UserID BIGINT, FirstName TEXT, SecondName TEXT, PhoneNumber TEXT, City TEXT, PostNumber INT);
CREATE TABLE OrderElement (ID INT PRIMARY KEY, Tea TEXT, Weight INT, Amount INT, OrderID INT, FOREIGN KEY(OrderID) REFERENCES Order(ID));
CREATE TABLE Order (ID INT PRIMARY KEY, OrderTime INT, UserId BIGINT, FOREIGN KEY(UserId) REFERENCES User(ID));
